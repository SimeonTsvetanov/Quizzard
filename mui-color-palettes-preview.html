üéØ How to Use These Colors in Your Theme

Method 1: Direct Hex Values

// In your App.tsx theme
palette: {
  mode,
  ...(mode === "light"
    ? {
        primary: { main: "#2196f3" }, // Material Blue
        secondary: { main: "#ff9800" }, // Material Orange
        // ...rest of your colors
      }
    : {
        primary: { main: "#64b5f6" }, // Lighter blue for dark mode
        secondary: { main: "#ffb74d" }, // Lighter orange for dark mode
        // ...rest of your colors
      }),
}

Method 2: Import MUI Colors (Recommended)

// At the top of App.tsx
import { blue, orange } from '@mui/material/colors';

// In your theme
palette: {
  mode,
  ...(mode === "light"
    ? {
        primary: { main: blue[500] }, // #2196f3
        secondary: { main: orange[500] }, // #ff9800
      }
    : {
        primary: { main: blue[300] }, // Lighter for dark mode
        secondary: { main: orange[300] }, // Lighter for dark mode
      }),
}
